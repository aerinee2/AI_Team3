// 1. 데이터 구조 정의 (JSON 변환용)
[System.Serializable]
public class FoodData {
    public string imageName;
    public string label;
}

[System.Serializable]
public class FoodDataWrapper {
    public List<FoodData> dataList = new List<FoodData>();
}

// 2. 실제 변환 및 저장 로직
void SaveAsJson(string fileName, string labelText) {
    string path = Path.Combine(Application.dataPath, "GeneratedData/labels.json");
    FoodDataWrapper wrapper = new FoodDataWrapper();

    // 기존 파일 있으면 읽어오기
    if (File.Exists(path)) {
        string existingJson = File.ReadAllText(path);
        wrapper = JsonUtility.FromJson<FoodDataWrapper>(existingJson);
    }

    // 새 데이터 추가
    wrapper.dataList.Add(new FoodData { imageName = fileName, label = labelText });

    // JSON 저장
    File.WriteAllText(path, JsonUtility.ToJson(wrapper, true));
}
